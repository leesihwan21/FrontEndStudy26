// Number (숫자) 타입 let studentAge = 20; let height = 175.5; let temperature = -5; console.log(studentAge); // 결과: 20 console.log(typeof studentAge); // 결과: "number" // String (문자열) 타입 let studentName = "김개발"; let greeting = '안녕하세요!'; let message = "오늘 온도는 " + temperature + "도입니다."; console.log(studentName); // 결과: 김개발 console.log(typeof studentName); // 결과: "string" // Boolean (불린) 타입 let isStudent = true; let isGraduated = false; console.log(isStudent); // 결과: true console.log(typeof isStudent); // 결과: "boolean" // Undefined 타입 let futureJob; console.log(futureJob); // 결과: undefined console.log(typeof futureJob); // 결과: "undefined" // Null 타입 let previousJob = null; console.log(previousJob); // 결과: null console.log(typeof previousJob); // 결과: "object" (JavaScript의 특이한 점!)
